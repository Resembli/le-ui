---
title: Variable Size Grid
---

Remember to import the `<Grid />` component. It is not shown in the examples below.

```js
import { Grid } from "@resembli/le-window"
```

For more on the `<Grid />` component, please refer to the component's [API documentation](/docs/le-window/grid)

## Basic Example

```js live
function GridVariableSizeBasicExample() {
  const gridData = Array(1000)
    .fill(0)
    .map((_, row) => {
      const cells = Array(200)
        .fill(0)
        .map((_, column) => {
          return { props: { row, column } }
        })

      return { cells }
    })

  const heights = [75, 100, 125, 100, 80]
  const widths = [80, 90, 100, 90, 80, 120]

  const gridHeights = Array(1000)
    .fill(0)
    .map((_, i) => heights[i % heights.length])

  const gridWidths = Array(200)
    .fill(0)
    .map((_, i) => widths[i % widths.length])

  function getRandomHEXColor() {
    const SEED = "0123456789abcdef"
    let output = "#"
    while (output.length < 7) {
      output += SEED[Math.floor(Math.random() * SEED.length)]
    }
    return output
  }

  const CellItem = ({ row, column }) => {
    const light = row % 2 === 1 ? "white" : "#f8f8f0"
    const dark = row % 2 === 1 ? "#f8f8f0" : "white"

    const color = row % 16 >= 7 ? "black" : "yellow"

    const startColor = useMemo(() => getRandomHEXColor(), [])
    const endColor = useMemo(() => getRandomHEXColor(), [])

    return (
      <div
        style={{
          display: "flex",
          justifyContent: "center",
          alignItems: "center",
          height: "100%",
          background: `linear-gradient(to right, ${startColor}, ${endColor}`,
          color: "black",
        }}
      >
        ({row}, {column})
      </div>
    )
  }

  return (
    <div style={{ height: 600 }}>
      <Grid
        data={gridData}
        defaultColumnWidth={100}
        defaultRowHeight={100}
        ItemComponent={CellItem}
        rowHeights={gridHeights}
        columnWidths={gridWidths}
      />
    </div>
  )
}
```
